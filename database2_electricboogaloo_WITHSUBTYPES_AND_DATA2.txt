
-- create the database
DROP DATABASE IF EXISTS murderhoboinc;
CREATE DATABASE murderhoboinc;

-- select the database
USE murderhoboinc;

-- create the tables
CREATE TABLE department
(
  dept_id   INT  PRIMARY KEY   AUTO_INCREMENT,
  dept_name     VARCHAR(50)    NOT NULL
);

CREATE TABLE factory
(
  factory_id      INT          PRIMARY KEY auto_increment,
  factory_address       VARCHAR(50)  NOT NULL,
  factory_city                   VARCHAR(50)    NOT NULL ,
  factory_state                  CHAR(2)        NOT NULL,
  factory_zip_code               VARCHAR(20)    NOT NULL
);

CREATE TABLE e_position
(
  pos_id              INT            PRIMARY KEY   AUTO_INCREMENT,
  pos_name     VARCHAR(50)    NOT NULL,
  pos_rateofpay	decimal(5,2),
  pos_dept int not null,
  constraint pos_fk_dept
	foreign key (pos_dept)
	references department(dept_id)
);

CREATE TABLE employee
(
  emp_id       INT            PRIMARY KEY auto_increment,
  emp_fname varchar(35) not null,
  emp_lname varchar(35) not null,
  emp_phone varchar(25),
  emp_email varchar(50),
  emp_address varchar(50) not null,
  emp_city varchar(25) not null,
  emp_state char(2) not null,
  emp_zip varchar(10) not null,
  emp_pos int not null,
  emp_factory int not null,
 CONSTRAINT emp_fk_pos
    FOREIGN KEY (emp_pos)
    REFERENCES e_position (pos_id),
constraint emp_fk_factory
    foreign key (emp_factory)
    references factory (factory_id) 
  
);




CREATE TABLE customer
(
  cust_id                     INT            PRIMARY KEY   AUTO_INCREMENT,
  cust_name                   VARCHAR(50)    NOT NULL,
  cust_address               VARCHAR(50),
  cust_city                   VARCHAR(50)    NOT NULL default 'Ababwa',
  cust_state                  CHAR(2)        NOT NULL,
  cust_zip_code               VARCHAR(20)    NOT NULL,
  cust_phone                  VARCHAR(50),
  cust_contact_last_name      VARCHAR(50),
  cust_contact_first_name     VARCHAR(50),
  cust_contact_email varchar(50)     
);

CREATE TABLE invoice
(
  invoice_id            INT            PRIMARY KEY   AUTO_INCREMENT,
  cust_id             INT            NOT NULL,
  invoice_date          DATE           NOT NULL,
  invoice_shipdate DATE,
  invoice_total decimal(9,2) default null,
  CONSTRAINT invoices_fk_customers
    FOREIGN KEY (cust_id)
    REFERENCES customer (cust_id)

);

create table product_type
(
	product_type_id int primary key auto_increment,
    product_type varchar(25)
);


create table product
(
	product_id int primary key auto_increment,
	product_type_id int not null,
    product_factory  int not null,
    product_totalqty int,
    constraint product_fk_factory
        foreign key (product_factory)
        references factory(factory_id),
	constraint product_fk_prodtype
		foreign key(product_type_id)
        references product_type(product_type_id)

);


create table product_order
(
	product_id int,
    invoice_id int,
    product_price decimal(9,2),
    product_qty int,
    order_total DECIMAL(9,2) default (product_price*product_qty),
	primary key(product_id, invoice_id),
    constraint product_id_fk_order
		foreign key (product_id)
        references product(product_id),
	constraint invoice_id_fk_order
		foreign key (invoice_id)
        references invoice(invoice_id)
    
);

create table di_set
(
	product_id int primary key,
    product_type_id  int not null default 1 check (product_type_id = 1),
    set_color varchar(25) not null,
    set_material varchar(25) not null,
    set_pcs int not null,
    constraint di_fk_id
		foreign key (product_id)
        references product(product_id),
	constraint di_fk_type
		foreign key (product_type_id)
        references product_type(product_type_id)
);

create table miniature
(
	product_id int primary key,
    product_type_id  int not null default 2 check (product_type_id = 2),
    mini_model varchar(25) not null,
    mini_size varchar(25) not null 
		check(mini_size = 'S' OR mini_size = 'M' OR mini_size='L'),
    painted bool,
    constraint mini_fk_id
		foreign key (product_id)
        references product(product_id),
	constraint mini_fk_type
		foreign key (product_type_id)
        references product_type(product_type_id)
);

create table spell_deck
(
	product_id int primary key,
    product_type_id  int not null default 3 check (product_type_id = 3),
    deck_class varchar(25) not null,
    constraint deck_fk_id
		foreign key (product_id)
        references product(product_id),
	constraint deck_fk_type
		foreign key (product_type_id)
        references product_type(product_type_id)
);
